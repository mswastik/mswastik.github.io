State wise GDP

library(ggplot2)
      library(plyr)
      ss=read.csv(file="data/cleaned/stategdp1.csv")
      gr=ddply(ss,c("Category","State.Union.Teritorries"),transform, Growth=c(NA,exp(diff(log(value.KCr)))-1))
      ggplot(data=gr, aes(x=State.Union.Teritorries, y=value.KCr, fill=year)) + geom_bar(stat="identity") + theme(axis.text.x = element_text(angle=70, hjust=1, face="bold"))
